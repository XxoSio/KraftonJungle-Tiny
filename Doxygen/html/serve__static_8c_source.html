65{
66 // filename 문자열 안에 '.html'이 있는지 검사

70 strcpy(filetype, "image/gif");
71 else if (strstr(filename, ".png"))
72 strcpy(filetype, "image/png");
73 else if (strstr(filename, ".jpg"))
74 strcpy(filetype, "image/jpeg");
75 else
76 strcpy(filetype, "text/plain");
