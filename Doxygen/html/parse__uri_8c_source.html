1/*
2 * parse_uri - parse URI into filename and CGI args
3 * return 0 if dynamic content, 1 if static
4 * - URI를 파일 이름 및 CGI 인수로 구문 분석
5 * 동적 내용인 경우 0을 반환하고 정적인 경우 1을 반환
6 *
7 */
8#include "csapp.h"
9
10// Tiny는 정적 컨텐츠를 위한 홈 디렉토리가 자신의 현재 디렉토리이고,
11// 실행파일의 홈 디렉토리는 /cgi-bin 이라고 가정
12int parse_uri(char *uri, char *filename, char *cgiargs)
13{
14 char *ptr;
15
16 // strstr(대상 문자열, 검색할 문자열) -> 검색된 문자열(대상 문자열)뒤에 모든 문자열이 나옴
17 // uri에서 'cgi-bin'이라는 문자열이 없으면 정적 컨텐츠
18 if (!strstr(uri, "cgi-bin")) {
19 // cgiargs 인자 string을 지움
20 strcpy(cgiargs, "");
21 // 상대 리눅스 경로 이름으로 변환 - '.'
22 strcpy(filename, ".");
23 // 상대 리눅스 경로 이름으로 변환 - '.' + '/index.html'
24 strcat(filename, uri);
25 // URI가 '/'문자로 끝나는 경우
26 if (uri[strlen(uri)-1] == '/')
27 // 기본 파일 이름인 home.html을 추가
28 strcat(filename, "home.html");
29
30 return 1;
31 }
32 // 동적 컨텐츠
33 else {
34 // 모든 CGI인자를 추출
35 ptr = index(uri, '?');
36
37 // index: 첫번째에서 두번째 인자를 찾음 - 찾으면 위치 포인터를, 못찾으면 NULL을 반환
38 if (ptr) {
39 strcpy(cgiargs, ptr+1);
40 *ptr = '\0';
41 }
42 // ?가 없는 경우
43 else
44 // 빈칸으로 둠
45 strcpy(cgiargs, "");
46
47 // 나머지 URI 부분을 상대 리눅스 파일이름으로 변환
48 strcpy(filename, ".");
49 strcat(filename, uri);
50
51 return 0;
52 }
53}
